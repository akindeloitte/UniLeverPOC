    page = st.sidebar.radio("Go to", [

    ])
    
    if page == "Executive Dashboard":
        show_executive_dashboard()
    elif page == "Incident Details":
        st.markdown('<div class="title-box"><h1>üîç INCIDENT DETAILS</h1></div>', unsafe_allow_html=True)
